//---------------------------------------------------------------------------
Stack* InStack(Stack *p, int in)
{
        Stack *t = new Stack;

        t -> info = in;
        t -> next = p;
        return t;
}

//---------------------------------------------------------------------------
void Del_All(Stack **p)
{
        while(*p != NULL)
                {
                t = *p;
                *p = (*p) -> next;
                delete t;
                }
}
