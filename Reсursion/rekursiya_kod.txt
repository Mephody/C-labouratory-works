double psqrtR(double s, int k)
        {
         s++;
         if (s<n)
                {
                 k++;
                 s=sqrt(k+psqrtR(s,k));
                 return s;
                }
                else
                        {
                         s=sqrt(n);
                         return s;
                        }
        }

double psqrt()
        {
         double f;
         f=sqrt(n);
         for(int i=n-1;i>0;i--) f=sqrt(i+f);
         return f;

        }

int ivvod()

         char s[100];

                while(true)

                                        if ((s[i]<'0' || s[i]>'9') && (s[i]!='-'))
                                                {

                                                 break;
                                                }
                                        i++;
                                        }

        }
